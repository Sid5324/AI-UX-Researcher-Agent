class AIModelManager:
    """
    Central manager for all AI providers.
    Implements fallback chain: Primary → Fallbacks → Error
    """
    
    def __init__(self):
        self.primary = OllamaClient(settings.ollama_model)
        self.fallbacks = [
            OllamaClient(model) 
            for model in settings.ollama_fallback_models.split(",")
        ]
        # Cloud providers (disabled until API keys provided)
        self.openrouter = None
        self.gemini = None
    
    async def generate_with_fallback(
        self,
        agent_type: str,  # "data", "prd", "ui_ux"
        prompt: str,
        system_prompt: Optional[str] = None,
        require_json: bool = False  # For structured outputs
    ) -> GenerationResult:
        """
        Generate with automatic fallback on failure.
        
        Agent-specific system prompts based on agent_type:
        - data: "You are a data analyst. Be precise, cite numbers."
        - prd: "You are a principal PM. Be structured, actionable."
        - ui_ux: "You are a staff designer. Be visual, detailed."
        """
        # Try primary
        result = await self.primary.generate(prompt, system_prompt)
        if result.success:
            return result
        
        # Try each fallback
        for client in self.fallbacks:
            result = await client.generate(prompt, system_prompt)
            if result.success:
                return result
        
        # All failed
        return GenerationResult(
            success=False,
            error="All AI models failed",
            model="none",
            provider="none",
            content=""
        )
Behavior:
Always tries local Ollama first (cost, privacy, latency)
Falls back through qwen2.5-coder, llava, etc.
Cloud providers (OpenRouter, Gemini) only if enabled and local fails
Tracks usage per agent for cost analysis
